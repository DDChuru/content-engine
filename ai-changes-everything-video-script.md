### Before/After Comparison

**Angular Component (Before):**
```typescript
// Angular - 45 lines, 4 files
@Component({
  selector: 'app-user-card',
  templateUrl: './user-card.component.html',
  styleUrls: ['./user-card.component.scss']
})
export class UserCardComponent implements OnInit, OnDestroy {
  @Input() user!: User;
  @Output() onSelect = new EventEmitter<User>();

  private destroy$ = new Subject<void>();
  isLoading = false;

  constructor(private userService: UserService) {}

  ngOnInit(): void {
    this.userService.status$
      .pipe(takeUntil(this.destroy$))
      .subscribe(status => this.handleStatus(status));
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  handleClick(): void {
    this.onSelect.emit(this.user);
  }
}
```

**React Equivalent (After):**
```tsx
// React - 20 lines, 1 file
interface UserCardProps {
  user: User;
  onSelect: (user: User) => void;
}

export function UserCard({ user, onSelect }: UserCardProps) {
  const [isLoading, setIsLoading] = useState(false);
  const status = useUserStatus();

  useEffect(() => {
    handleStatus(status);
  }, [status]);

  return (
    <div className="user-card" onClick={() => onSelect(user)}>
      {/* template */}
    </div>
  );
}
```

**SvelteKit Equivalent (After):**
```svelte
<!-- Svelte - 15 lines, 1 file -->
<script lang="ts">
  import { userStatus } from '$lib/stores/user';

  export let user: User;
  export let onSelect: (user: User) => void;

  let isLoading = false;

  $: handleStatus($userStatus);
</script>

<div class="user-card" on:click={() => onSelect(user)}>
  <!-- template -->
</div>
```
